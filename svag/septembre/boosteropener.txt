'use client'

import { useState } from 'react'
import { generateBooster } from '@/lib/generateBooster'
import type { Card } from '@/types/card'

export default function BoosterOpener() {
  const [booster, setBooster] = useState<Card[]>([])
  const [opened, setOpened] = useState(false)

  const handleOpen = async () => {
    const cards = await generateBooster()
    setBooster(cards)
    setOpened(true)
  }

  return (
    <div className="flex flex-col items-center gap-6">
      {!opened ? (
        <button
          onClick={handleOpen}
          className="px-6 py-3 bg-yellow-500 text-black font-bold rounded hover:bg-yellow-600 transition"
        >
          üéÅ Ouvrir un booster
        </button>
      ) : (
        <div className="grid grid-cols-2 md:grid-cols-3 gap-6">
          {booster.map((card, index) => (
            <div
              key={index}
              className={`booster-card p-4 rounded-lg border shadow-lg relative ${
                card.rarity === 'legendary'
                  ? 'border-yellow-400 animate-foil'
                  : card.rarity === 'epic'
                  ? 'border-purple-500'
                  : 'border-gray-300'
              }`}
            >
              <img
                src={card.image_url}
                alt={card.name}
                className="w-full h-auto rounded"
              />
              <div className="mt-2 text-center font-semibold text-sm">
                {card.make ?? card.name} {card.model ?? ''} ({card.year})
              </div>
              <div className="text-xs text-gray-500 text-center">
                {card.type} ‚Ä¢ {card.rarity}
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  )
}
